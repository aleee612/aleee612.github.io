> 顾名思义，这类题目的做法是使用一个单调递增/递减的栈，来实现大小的对比
> 由此可以将时间复杂度控制在 $O(n)$ ，而非一个个比较的 $O(n^2)$ 
### 练习题
##### p1318 积水面积
![[Pasted image 20260310203354.png|490]]
```cpp fold
#include<iostream>
#include<math.h>
#include<stack>
using namespace std;
int a[10000];
int c[2][10000];

int main() {
	int n;
	stack<int> b;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	for (int i = 0; i < n; i++) {
		if (b.empty()||a[i] < a[b.top()]) {
			b.push(i);// 栈为空 或 小的入栈
		}
		else{
			int size = b.top();
			c[1][size] = i; // 遇到大或相等的出栈，c1列为右大
			b.pop();
			if (b.empty()) {
				c[0][size] = -1;
			}
			else
				c[0][size] = b.top(); // c0列为左大
			i--; // 继续比较当前元素
		}
	}
	while (!b.empty()) {
		int size = b.top();
		c[1][size] = -1; // 右无大
		b.pop();
		if (b.empty()) {
			c[0][size] = -1; // 左无大
		}
		else
			c[0][size] = b.top(); // c0列为左大
	}
	int ans = 0;
	for (int i = 0; i < n; i++) {
		if (c[0][i] != -1 && c[1][i] != -1)
		{
			int left = c[0][i];
			int right = c[1][i];
			int height = min(a[left], a[right]) - a[i]; // 高度差
			int width = right - left - 1;              // 下标差
			if (height > 0 && width > 0)
				ans += 1LL * height * width;           // 防止溢出
		}
	}
	cout << ans;
	return 0;
}
```
- 虽然左程云推荐使用数组栈效果更佳，但那样代码写的过于复杂容易出错，因此还是使用cpp自带栈实现
- 核心思路与单调栈经典例题无差，主要在于获取每个元素左右的第一个高于其的元素下标位置，便能计算出当前雨水覆盖面积
- 需注意最终计算面积时要考虑高宽是否大于0
##### P2422 良好的感觉
![[Pasted image 20260311150820.png|529]]
```cpp fold
#include<iostream>
#include<stack>
#include<math.h>
using namespace std;
long long a[100001];
long long b[2][100001];
long long pre[100001];
int main()
{
	int n;
	stack<long long> s;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	pre[0] = a[0];
	for (int i = 1; i < n; i++) {
		pre[i] = pre[i - 1] + a[i];
	}
	for (int i = 0; i < n; i++)
	{
		if (s.empty() || a[i] > a[s.top()]) {
			s.push(i);
		}
		else {
			int top = s.top();
			b[1][top] = i;
			s.pop();
			if (s.empty()) {
				b[0][top] = -1;
			}
			else
				b[0][top] = s.top();
			i--; // 继续比较当前元素
		}
	}
	while (!s.empty()) {
		int top = s.top();
		b[1][top] = n;
		s.pop();
		if (s.empty()) 
			b[0][top] = -1;
		else
			b[0][top] = s.top();
	}
	long long Max = 0;
	for (int i = 0; i < n; i++) {
		int x = b[0][i]+1;
		int y = b[1][i]-1;
		long long sum;
		if (x == 0)
			sum = pre[y] * a[i];
		else
			sum = (pre[y] - pre[x - 1]) * a[i];
		if (y >= x) {
			//cout << i << " " << b[0][i] << " " << b[1][i] << endl;
			Max = max(Max, sum);
		}
	}
	cout << Max;
	return 0;
}
```
- 主要思路为单调栈**大压小**，后续处理时使用**前缀和**压缩累计总和时间。
- 根据大压小的特性，遇到比自己小/相等的元素弹出，从而使在xy范围内，当前元素为最小值。在该情况下，计算出的舒适值最大。
- 再去计算每一个元素的最大值，获取出最大的结果。
- 左右元素范围的划分：
	- 对于当前左侧无比自己小的值，将左值划为-1
	- 对于右侧，则划为n
	- 这样在最终xy范围的划分时可以只得出比自己大的元素
