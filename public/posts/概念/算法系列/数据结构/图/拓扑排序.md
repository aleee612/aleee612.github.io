> 拓扑排序：保证顺应边的顺序即可，拓扑排序可以有多个结果
#### 核心思想：
1. 计算所有节点的入度、出度
2. 将所有入度为0的节点删去，并清除其影响：即与该节点相连的节点入度变为0
3. 循环2步骤，直至所有节点都被删去
- 为达成步骤2，需要使用邻接表记录整个图结构

```cpp fold
class Solution {
public:
    vector<int> findOrder(int numCourses, vector<vector<int>>& prerequisites) {
        vector<vector<int>> map(numCourses);
        int pre[2000] = {0};
        for (int i = 0; i < prerequisites.size(); i++) {
            map[prerequisites[i][1]].push_back(prerequisites[i][0]);
            pre[prerequisites[i][0]]++;
        }
        vector<int> ans;
        int q[2000];
        int l = 0, r = 0;
        for (int i = 0; i < numCourses; i++) {
            if (pre[i] == 0) { // 入度为0
                q[r++] = i;    // 入队列
            }
        }
        while (l < r) { // 队列不为空时
            int cur = q[l];
            ans.push_back(q[l]);
            for (int j = 0; j < map[cur].size(); j++) {
                pre[map[cur][j]]--; // 对应的下一位的入度减一
                if (pre[map[cur][j]] == 0) { // 减一后当前元素对应的入度若为0
                    q[r++] = map[cur][j]; // 就入栈
                }
            }
            l++; // 清理完当前元素后，将该元素剔除
        }
        if (ans.size() != numCourses) return {};
        return ans;
    }
};
```
- 注意事项：while循环中不使用for来进行对队列的遍历，而是用对于 `l` 的控制来实现
- 老问题，需注意元素是值还是下标

```cpp fold
#include<iostream>
using ll = long long;
using namespace std;
	ll a[500001][2];
	ll head[5001];
	ll next1[500001];
	ll to[500001];
	int main() {
		ll n, m;
		cin >> n >> m;
		for (ll i = 0; i < m;i++) {
			cin >> a[i][0] >> a[i][1];
		}
		ll cnt = 1;
		for (ll i = 0; i < m; i++) {
			ll u = a[i][0]; // 出发点
			ll v = a[i][1]; // 目的地
			next1[cnt] = head[u]; // 边的下一条边
			head[u] = cnt; // 头边号
			to[cnt] = v; // 边指向的下一个节点
			cnt++;
		}
		ll inde[5001] = { 0 }; // 入度表
		ll add[5001] = { 0 }; // 加权表
		for (ll i = 1; i <= m; i++) {
			inde[to[i]]++; // 统计入度
		}
		for (ll i = 1; i <= n; i++) {
			if (inde[i] == 0) {
				add[i] = 1; // 入度为0的节点权值为1
			}
		}
		ll q[5001];
		ll l = 1, r = 1;
		for (ll i = 1; i <= n; i++) {
			if (add[i] == 1) {
				q[r++] = i; // 入度为0的节点入队
			}
		}
		ll ans = 0;
		while(l != r){
			ll u = q[l++]; // 当前节点
			if (head[u] == 0) {
				ans = (ans + add[u]) % 80112002; // 累加权值
			}
			else { // 当 当前 u 节点的 head 值不为0时，说明该链上还有元素，则往下继续遍历，直至到末尾元素为止 （head[u] == 0）
				for (ll i = head[u]; i != 0;i = next1[i]) { // 往下遍历，遍历的方向是next1数组中对应存储的下标
					ll v = to[i]; // 下一个节点，用 v 存一下
					add[v] = (add[v] + add[u]) % 80112002; // 将v的权值与上一个节点u的累加，达到计算几条路的效果
					if(--inde[v] == 0) { // 操作完v，将其入度减1，说明u此时已经清除了对他的影响
						q[r++] = v; // 如果此时 v 入度为 0 了，将节点入队
					}
				}
			}
		}
		cout << ans << endl;
		return 0;
	}
```
- 由于数据量过大，使用链式前向星存储图
- 核心思想：
	1. 使用加权表记录当前有几个路径，当当前节点入度为0时，累加权值
	2. 当当前节点的head值为0时：即为最后末尾元素，直接用ans累加权值
	3. 继续将接下来入度为0的点入队列
