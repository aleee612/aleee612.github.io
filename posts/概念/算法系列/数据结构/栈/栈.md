### cpp STL
```cpp fold
#include <iostream>
#include <stack>
using namespace std;

int main() {
    stack<int> s;          // 定义栈
    s.push(1);             // 入栈
    s.top();               // 访问栈顶元素，但不会出栈
    s.pop();               // 出栈（删除栈顶）
    s.empty();             // 判断是否为空 (true / false)
    s.size();              // 栈元素个数
    return 0;
}
```
